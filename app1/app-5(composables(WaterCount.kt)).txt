package com.qi.aulamanha5.composables

import androidx.compose.foundation.layout.*
import androidx.compose.material3.Button
import androidx.compose.material3.Divider
import androidx.compose.material3.Text
import androidx.compose.runtime.Composable
import androidx.compose.runtime.getValue
import androidx.compose.runtime.setValue
import androidx.compose.runtime.mutableStateOf
import androidx.compose.runtime.remember
import androidx.compose.ui.Modifier
import androidx.compose.ui.res.stringResource
import androidx.compose.ui.unit.dp

@Composable
fun WaterCountState(
    modifier: Modifier = Modifier
) {
    var counter by remember {
        mutableStateOf(0)
    }

    AulaManha5(
        count = counter,
        onIncrement = { counter++ },
        onReset = { counter = 0 }
    )
}

    @Composable
    fun AulaManha5(
        modifier: Modifier = Modifier,
        count: Int,
        onIncrement: () -> Unit,
        onReset: () -> Unit
    ) {
        Column(
            modifier = modifier.fillMaxWidth()
        ) {
            Row(
                modifier = modifier.fillMaxWidth(),
                horizontalArrangement = Arrangement.SpaceAround
            ) {
                //Comtera os dois bot√µes
                Button(
                    onClick = { onIncrement },
                    enabled = count < 10
                ) {
                    Text(
                        text = stringResource(id = com.qi.aulamanha5.R.string.botao1)
                    )

                }
                Button(
                    onClick = { onReset },
                    enabled = count > 0
                ) {
                    Text(
                        text = stringResource(id = com.qi.aulamanha5.R.string.botao2)
                    )
                }

            }

            Divider(
                modifier = modifier.padding(10.dp),
                thickness = 2.dp
            )

            Text(
                text = stringResource(id = com.qi.aulamanha5.R.string.botao3) + count
            )

        }
    }
